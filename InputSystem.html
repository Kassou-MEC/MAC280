C# Switching Action Maps 
// Get reference to your input actions
private PlayerInputActions inputActions;

void EnterVehicle()
{
    // Disable player controls
    inputActions.Player.Disable();
    
    // Enable vehicle controls
    inputActions.Vehicle.Enable();
}

{
    inputActions.Vehicle.Disable();
    inputActions.Player.Enable();
}

using UnityEngine;
using UnityEngine.InputSystem;

public class PlayerMovement : MonoBehaviour
{
    public float moveSpeed = 5f;
    private Vector2 moveInput;

    // Called by Player Input component
    public void OnMove(InputValue value)
    {
        moveInput = value.Get<Vector2>();
    }

    void Update()
    {
        Vector3 movement = new Vector3(moveInput.x, 0, moveInput.y);
        transform.Translate(movement * moveSpeed * Time.deltaTime);
    }
}

C# PlayerController.cs - Complete Example 
using UnityEngine;
using UnityEngine.InputSystem;

public class PlayerController : MonoBehaviour
{
    public float moveSpeed = 5f;
    public float jumpForce = 10f;
    
    private Vector2 moveInput;
    private Rigidbody rb;

    void Start()
    {
        rb = GetComponent<Rigidbody>();
    }

    // Called automatically when "Move" action is performed
    public void OnMove(InputValue value)
    {
        moveInput = value.Get<Vector2>();
    }

    // Called automatically when "Jump" action is performed
    public void OnJump(InputValue value)
    {
        if (value.isPressed)
        {
            rb.AddForce(Vector3.up * jumpForce, ForceMode.Impulse);
        }
    }

    void FixedUpdate()
    {
        Vector3 movement = new Vector3(moveInput.x, 0, moveInput.y);
        rb.MovePosition(rb.position + movement * moveSpeed * Time.fixedDeltaTime);
    }
}
Method Naming Convention

Method names must follow the pattern On[ActionName] and the script must be on the same GameObject as the Player Input component!

C# Script for Unity Events 
using UnityEngine;
using UnityEngine.InputSystem;

public class PlayerActions : MonoBehaviour
{
    // Method signature must include InputAction.CallbackContext
    public void HandleMove(InputAction.CallbackContext context)
    {
        Vector2 input = context.ReadValue<Vector2>();
        Debug.Log($"Moving: {input}");
    }

    public void HandleJump(InputAction.CallbackContext context)
    {
        // Check if this is the "performed" phase (button pressed)
        if (context.performed)
        {
            Debug.Log("Jump!");
        }
    }
}

Advantages

  • Visual setup in Inspector
  • Designers can modify without code
  • Methods can be on any GameObject
  • Multiple functions per action

Considerations

  • Requires CallbackContext parameter
  • Check context.phase for timing
  • Serialized references can break
  • Harder to debug connections

C# Events (Advanced)

Maximum control with event subscriptions

When to Use C# Events

This approach gives you complete programmatic control. Best for complex games, multiplayer systems, or when you need dynamic event handling.

C# Advanced C# Events Pattern 
using UnityEngine;
using UnityEngine.InputSystem;

public class AdvancedPlayerController : MonoBehaviour
{
    private PlayerInput playerInput;
    private InputAction moveAction;
    private InputAction jumpAction;

    void Awake()
    {
        playerInput = GetComponent<PlayerInput>();
        
        // Get action references
        moveAction = playerInput.actions["Move"];
        jumpAction = playerInput.actions["Jump"];
    }

    void OnEnable()
    {
        // Subscribe to events
        jumpAction.performed += OnJumpPerformed;
        jumpAction.canceled += OnJumpCanceled;
    }

    void OnDisable()
    {
        // Always unsubscribe to prevent memory leaks!
        jumpAction.performed -= OnJumpPerformed;
        jumpAction.canceled -= OnJumpCanceled;
    }

    private void OnJumpPerformed(InputAction.CallbackContext ctx)
    {
        Debug.Log("Jump started!");
    }

    private void OnJumpCanceled(InputAction.CallbackContext ctx)
    {
        Debug.Log("Jump released!");
    }

    void Update()
    {
        // Read continuous input directly
        Vector2 move = moveAction.ReadValue<Vector2>();
        transform.Translate(new Vector3(move.x, 0, move.y) * Time.deltaTime);
    }
}
Critical: Always Unsubscribe!

If you subscribe to events in OnEnable(), you must unsubscribe in OnDisable(). Failing to do so causes memory leaks and duplicate event calls!

Action Phases

Phase When It Fires Use Case
started Input begins Start charging attack
performed Input completes/threshold met Execute action
canceled Input releases/interrupted Stop action, release charge

C# Using the Generated Class 
using UnityEngine;

public class PlayerWithGeneratedClass : MonoBehaviour
{
    // The generated class - no string references needed!
    private PlayerInputActions inputActions;

    void Awake()
    {
        inputActions = new PlayerInputActions();
    }

    void OnEnable()
    {
        inputActions.Enable();
        
        // Strongly-typed access to actions!
        inputActions.Player.Jump.performed += ctx => Jump();
    }

    void OnDisable()
    {
        inputActions.Disable();
    }

    void Update()
    {
        // Direct access - IDE autocomplete works!
        Vector2 move = inputActions.Player.Move.ReadValue<Vector2>();
        
        // No more typos in action names
        if (inputActions.Player.Fire.WasPerformedThisFrame())
        {
            Fire();
        }
    }

    void Jump() { /* Jump logic */ }
    void Fire() { /* Fire logic */ }
}

Benefits

  • No magic strings
  • Full IDE autocomplete
  • Compile-time error checking
  • Strongly typed values

Remember

  • Regenerate after changes
  • Must call Enable() manually
  • Create new instance in Awake
  • Dispose/Disable when done
