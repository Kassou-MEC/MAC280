C# WASDMovementBasic.cs 
using UnityEngine;

public class WASDMovementBasic : MonoBehaviour
{
    // Public variable - visible in Inspector, controls movement speed
    public float moveSpeed = 5f;

    // Update is called once per frame
    void Update()
    {
        // Check if W key is being held down
        if (Input.GetKey(KeyCode.W))
        {
            // Move upward: Vector2.up = (0, 1)
            transform.Translate(Vector2.up * moveSpeed * Time.deltaTime);
        }
        // Check if S key is being held down
        if (Input.GetKey(KeyCode.S))
        {
            // Move downward: Vector2.down = (0, -1)
            transform.Translate(Vector2.down * moveSpeed * Time.deltaTime);
        }
        // Check if A key is being held down
        if (Input.GetKey(KeyCode.A))
        {
            // Move left: Vector2.left = (-1, 0)
            transform.Translate(Vector2.left * moveSpeed * Time.deltaTime);
        }
        // Check if D key is being held down
        if (Input.GetKey(KeyCode.D))
        {
            // Move right: Vector2.right = (1, 0)
            transform.Translate(Vector2.right * moveSpeed * Time.deltaTime);
        }
    }
}
Code Explanation - Line by Line
  • public float moveSpeed = 5f; - Declares a public variable visible in Unity Inspector. The 'f' suffix indicates a float (decimal) number. Value of 5 means 5 units per second.
  • void Update() - Unity's built-in method called every frame (60 times per second at 60 FPS). Used for input detection and non-physics movement.
  • Input.GetKey(KeyCode.W) - Checks if the W key is currently pressed. Returns true every frame while held down, false when released.
  • transform.Translate() - Moves the GameObject relative to its current position. Takes a Vector (direction and distance) as parameter.
  • Vector2.up * moveSpeed * Time.deltaTime - Calculates movement: direction (0,1) √ó speed (5) √ó frame time (~0.016) = smooth, frame-rate independent movement.
Important: Time.deltaTime

Without Time.deltaTime: Movement would be tied to frame rate (faster on powerful PCs, slower on weak ones)

With Time.deltaTime: Movement is consistent regardless of frame rate - essential for fair gameplay!

C# ArrowKeysMovement.cs 
using UnityEngine;

public class ArrowKeysMovement : MonoBehaviour
{
    // Movement speed in units per second
    public float moveSpeed = 5f;
    
    // Private variable to store movement direction
    private Vector2 movement;

    void Update()
    {
        // Reset movement to zero each frame
        movement = Vector2.zero;

        // Vertical movement - Up Arrow
        if (Input.GetKey(KeyCode.UpArrow))
        {
            movement.y = 1f;  // Set Y to positive (up)
        }
        // Vertical movement - Down Arrow
        if (Input.GetKey(KeyCode.DownArrow))
        {
            movement.y = -1f;  // Set Y to negative (down)
        }
        // Horizontal movement - Left Arrow
        if (Input.GetKey(KeyCode.LeftArrow))
        {
            movement.x = -1f;  // Set X to negative (left)
        }
        // Horizontal movement - Right Arrow
        if (Input.GetKey(KeyCode.RightArrow))
        {
            movement.x = 1f;  // Set X to positive (right)
        }

        // Normalize to prevent faster diagonal movement
        movement = movement.normalized;

        // Apply the movement to transform
        transform.Translate(movement * moveSpeed * Time.deltaTime);
    }
}
Code Explanation - Purpose of Each Section
  • private Vector2 movement; - Private variable (not visible in Inspector) stores the combined movement direction as x,y coordinates.
  • movement = Vector2.zero; - Resets movement to (0,0) each frame. Without this, previous frame's values would persist and cause issues.
  • movement.y = 1f; - Sets only the Y component. Using separate if statements (not else if) allows diagonal movement by setting both x and y.
  • movement.normalized - Critical for fair gameplay! Without this, diagonal movement (1,1) has magnitude 1.41, making diagonal movement 41% faster than cardinal directions.
Why Normalize? - The Math

Without normalization: Moving Up+Right = (1, 1), magnitude = ‚àö(1¬≤ + 1¬≤) = ‚àö2 ‚âà 1.414

With normalization: (1, 1).normalized = (0.707, 0.707), magnitude = 1.0

Result: Consistent speed in ALL directions!

Combined Input

WASD + Arrow Keys Combined

Support both input methods for maximum player accessibility

Key Definitions
  • || (OR operator) - Logical operator that returns true if EITHER condition is true
  • else if - Executes only if previous condition was false, prevents conflicting inputs
  • Input Priority - When opposite keys pressed, the first checked key wins
Purpose of This Script
  • Allow players to use their preferred control scheme
  • Support both WASD (left hand) and Arrow keys (right hand) simultaneously
  • Handle conflicting inputs gracefully (e.g., pressing W and S together)
  • Provide consistent, normalized movement speed
C# CombinedMovement.cs 
using UnityEngine;

public class CombinedMovement : MonoBehaviour
{
    // === VARIABLES ===
    public float moveSpeed = 5f;      // Movement speed (units/sec)
    private Vector2 moveDirection;    // Stores final movement vector

    void Update()
    {
        // Reset direction each frame
        moveDirection = Vector2.zero;

        // === VERTICAL MOVEMENT ===
        // Check W key OR Up Arrow for moving up
        if (Input.GetKey(KeyCode.W) || Input.GetKey(KeyCode.UpArrow))
        {
            moveDirection.y = 1f;
        }
        // Check S key OR Down Arrow for moving down
        // Using 'else if' prevents both up and down activating together
        else if (Input.GetKey(KeyCode.S) || Input.GetKey(KeyCode.DownArrow))
        {
            moveDirection.y = -1f;
        }

        // === HORIZONTAL MOVEMENT ===
        // Check A key OR Left Arrow for moving left
        if (Input.GetKey(KeyCode.A) || Input.GetKey(KeyCode.LeftArrow))
        {
            moveDirection.x = -1f;
        }
        // Check D key OR Right Arrow for moving right
        else if (Input.GetKey(KeyCode.D) || Input.GetKey(KeyCode.RightArrow))
        {
            moveDirection.x = 1f;
        }

        // === APPLY MOVEMENT ===
        // Normalize for consistent diagonal speed
        moveDirection = moveDirection.normalized;
        
        // Move the player
        transform.Translate(moveDirection * moveSpeed * Time.deltaTime);
    }
}
Code Explanation - Key Concepts
  • Input.GetKey(KeyCode.W) || Input.GetKey(KeyCode.UpArrow) - The || (OR) operator means: "if W is pressed OR if UpArrow is pressed, execute this code". Either input works!
  • else if for opposite directions - Using else if instead of if means that pressing W+S together will only move UP (W is checked first). This prevents the player from getting stuck.
  • Separate if blocks for vertical/horizontal - Notice vertical uses if/else if, and horizontal uses separate if/else if. This allows diagonal movement (pressing W+D moves up-right).
Pro Tip: Control Scheme Design

WASD: Preferred by most PC gamers, leaves right hand free for mouse

Arrow Keys: More intuitive for beginners and casual players

Best Practice: Always support both for maximum accessibility!

C# SmoothMovement.cs 
using UnityEngine;

public class SmoothMovement : MonoBehaviour
{
    // === CONFIGURATION ===
    public float moveSpeed = 5f;          // Base movement speed
    
    // === PRIVATE VARIABLES ===
    private float horizontalInput;        // Stores left/right input (-1 to 1)
    private float verticalInput;          // Stores up/down input (-1 to 1)
    private Vector2 moveVector;           // Combined movement direction

    void Update()
    {
        // === GET INPUT ===
        // GetAxis returns smooth values from -1 to 1
        // Automatically handles: A/D, Left/Right Arrow, Gamepad
        horizontalInput = Input.GetAxis("Horizontal");
        verticalInput = Input.GetAxis("Vertical");

        // === CREATE MOVEMENT VECTOR ===
        moveVector = new Vector2(horizontalInput, verticalInput);
        
        // === CLAMP MAGNITUDE ===
        // Ensures diagonal movement isn't faster than cardinal
        moveVector = Vector2.ClampMagnitude(moveVector, 1f);

        // === APPLY MOVEMENT ===
        transform.Translate(moveVector * moveSpeed * Time.deltaTime);
    }
}
Code Explanation - Why GetAxis is Superior
  • Input.GetAxis("Horizontal") - Unity's Input Manager automatically handles A, D, LeftArrow, RightArrow, AND gamepad left stick! One line replaces multiple GetKey checks.
  • Smooth values -1 to 1 - When you press D, value gradually increases: 0 ‚Üí 0.3 ‚Üí 0.6 ‚Üí 0.9 ‚Üí 1.0. This creates smooth acceleration instead of instant full-speed movement.
  • Vector2.ClampMagnitude(moveVector, 1f) - Different from normalized! ClampMagnitude allows values less than 1 (for analog stick partial tilt) but caps maximum at 1.
Method Returns Use Case
GetAxis() Smooth -1 to 1 Natural, smooth movement with acceleration
GetAxisRaw() Instant -1, 0, or 1 Precise, snappy controls (fighting games, platformers)
Physics-Based

Rigidbody2D Movement

Proper physics integration for collision detection and realistic movement

Key Definitions
  • Rigidbody2D - Component that makes GameObject respond to physics (gravity, forces, collisions)
  • rb.velocity - The current speed and direction of the Rigidbody (Vector2)
  • rb.MovePosition() - Moves Rigidbody to position while respecting physics
  • FixedUpdate() - Called at fixed intervals (default 50x/sec), used for physics calculations
  • GetComponent<T>() - Retrieves a component of type T attached to the GameObject
Why Use Rigidbody2D Instead of Transform?

Transform.Translate Problem: Moves object directly, ignoring physics. Objects can pass through walls!

Rigidbody2D Solution: Uses Unity's physics engine for proper collision detection and response.

C# PhysicsMovement.cs 
using UnityEngine;

public class PhysicsMovement : MonoBehaviour
{
    // === CONFIGURATION ===
    public float moveSpeed = 5f;      // Movement speed
    
    // === COMPONENT REFERENCES ===
    private Rigidbody2D rb;            // Reference to Rigidbody2D component
    
    // === INPUT STORAGE ===
    private Vector2 movement;          // Stores movement direction

    void Start()
    {
        // Cache the Rigidbody2D component for better performance
        // GetComponent is expensive - only call once in Start!
        rb = GetComponent<Rigidbody2D>();
    }

    void Update()
    {
        // === INPUT DETECTION ===
        // Read input in Update (runs every frame)
        // GetAxisRaw for instant, responsive controls
        movement.x = Input.GetAxisRaw("Horizontal");
        movement.y = Input.GetAxisRaw("Vertical");
        
        // Normalize for consistent diagonal speed
        movement = movement.normalized;
    }

    void FixedUpdate()
    {
        // === PHYSICS MOVEMENT ===
        // Apply movement in FixedUpdate (fixed physics timestep)
        // Method 1: Direct velocity assignment
        rb.velocity = movement * moveSpeed;
        
        // Alternative Method 2: MovePosition (more precise)
        // rb.MovePosition(rb.position + movement * moveSpeed * Time.fixedDeltaTime);
    }
}
Code Explanation - Physics Concepts
  • GetComponent<Rigidbody2D>() - Finds and returns the Rigidbody2D attached to this GameObject. Called once in Start() and stored in 'rb' variable for performance.
  • Update() vs FixedUpdate() - Update() runs every frame (variable rate). FixedUpdate() runs at fixed intervals (default 0.02 seconds = 50 times/sec). Physics must be in FixedUpdate for consistency!
  • rb.velocity = movement * moveSpeed - Directly sets the Rigidbody's velocity. Physics engine handles collision detection automatically.
  • rb.MovePosition() - Alternative method that moves toward a position. Better for precise movement but requires Time.fixedDeltaTime.
Update vs FixedUpdate - Critical Difference

Update(): Called every frame (60x/sec at 60 FPS, 144x/sec at 144 FPS). Use for INPUT detection.

FixedUpdate(): Called at fixed intervals (50x/sec by default). Use for PHYSICS calculations.

Why it matters: Physics calculations need consistent timing. Variable frame rates in Update() cause jittery physics!

Spacebar Jump

Implement jumping mechanics with GetKeyDown and ground detection

Key Definitions
  • Input.GetKeyDown() - Returns TRUE only on the FIRST frame the key is pressed
  • Input.GetKey() - Returns TRUE every frame while the key is HELD down
  • Input.GetKeyUp() - Returns TRUE only on the frame the key is RELEASED
  • isGrounded - Boolean flag tracking whether player is touching the ground
  • jumpForce - The upward velocity applied when jumping

C# JumpMovement.cs 
using UnityEngine;

public class JumpMovement : MonoBehaviour
{
    // === MOVEMENT SETTINGS ===
    public float moveSpeed = 5f;       // Horizontal movement speed
    public float jumpForce = 10f;      // Upward force when jumping
    
    // === STATE TRACKING ===
    public bool isGrounded = true;    // Is player on ground?
    
    // === COMPONENTS ===
    private Rigidbody2D rb;            // Physics component
    private Vector2 movement;          // Movement direction

    void Start()
    {
        // Get and store Rigidbody2D reference
        rb = GetComponent<Rigidbody2D>();
    }

    void Update()
    {
        // === HORIZONTAL INPUT ===
        movement.x = Input.GetAxisRaw("Horizontal");

        // === JUMP INPUT ===
        // GetKeyDown = only triggers ONCE when key is first pressed
        // && isGrounded = only allow jump when on ground
        if (Input.GetKeyDown(KeyCode.Space) && isGrounded)
        {
            Jump();
        }
    }

    void FixedUpdate()
    {
        // Apply horizontal movement, preserve vertical velocity (gravity)
        rb.velocity = new Vector2(movement.x * moveSpeed, rb.velocity.y);
    }

    void Jump()
    {
        // Apply upward velocity for jump
        // Preserve horizontal velocity (movement.x), set vertical to jumpForce
        rb.velocity = new Vector2(rb.velocity.x, jumpForce);
        
        // Mark player as not grounded (in air)
        isGrounded = false;
    }

    // === GROUND DETECTION (Simple Method) ===
    void OnCollisionEnter2D(Collision2D collision)
    {
        // When player touches something, they're grounded
        isGrounded = true;
    }
}
Code Explanation - Jump Mechanics
  • Input.GetKeyDown(KeyCode.Space) - Returns true ONLY on the first frame Space is pressed. Unlike GetKey, holding Space won't trigger multiple jumps.
  • && isGrounded - The AND operator ensures BOTH conditions must be true: Space pressed AND player on ground. Prevents infinite jumping in mid-air!
  • new Vector2(rb.velocity.x, jumpForce) - Creates a new velocity: keeps current horizontal speed (rb.velocity.x) but sets vertical to jumpForce. This allows jumping while moving.
  • OnCollisionEnter2D() - Unity automatically calls this when Rigidbody2D touches another collider. Simple ground detection method.
Method When It Returns TRUE Use Case
GetKeyDown() First frame key is pressed Jump, shoot, interact - single actions
GetKey() Every frame while held Movement, charging attacks
GetKeyUp() Frame key is released Release charged attack, stop action

Full Platformer Controller

Production-ready script with proper ground detection using Physics2D

Key Definitions
  • Physics2D.OverlapCircle() - Checks if any colliders overlap a circular area
  • LayerMask - Specifies which physics layers to include in detection
  • Transform groundCheck - Empty GameObject positioned at player's feet for ground detection
  • [Header("text")] - Attribute that adds a header label in the Unity Inspector
C# PlatformerController.cs 
using UnityEngine;

public class PlatformerController : MonoBehaviour
{
    // === MOVEMENT SETTINGS ===
    [Header("Movement")]
    public float moveSpeed = 7f;       // Horizontal speed
    public float jumpForce = 12f;      // Jump power
    
    // === GROUND DETECTION SETTINGS ===
    [Header("Ground Check")]
    public Transform groundCheck;      // Position to check for ground
    public float groundRadius = 0.2f; // Size of detection circle
    public LayerMask groundLayer;      // Which layers count as ground
    
    // === PRIVATE VARIABLES ===
    private Rigidbody2D rb;
    private bool isGrounded;
    private float horizontalInput;

    void Start()
    {
        // Cache Rigidbody2D reference
        rb = GetComponent<Rigidbody2D>();
    }

    void Update()
    {
        // === INPUT ===
        horizontalInput = Input.GetAxisRaw("Horizontal");
        
        // === GROUND CHECK ===
        // Creates invisible circle at groundCheck position
        // Returns true if circle overlaps any collider on groundLayer
        isGrounded = Physics2D.OverlapCircle(
            groundCheck.position,   // Center of circle
            groundRadius,           // Radius of circle
            groundLayer             // Which layers to check
        );
        
        // === JUMP ===
        if (Input.GetKeyDown(KeyCode.Space) && isGrounded)
        {
            rb.velocity = new Vector2(rb.velocity.x, jumpForce);
        }
    }

    void FixedUpdate()
    {
        // === MOVEMENT ===
        rb.velocity = new Vector2(horizontalInput * moveSpeed, rb.velocity.y);
    }
}
Code Explanation - Advanced Concepts
  • [Header("Movement")] - Unity Inspector attribute that creates a bold header above the following variables. Makes Inspector organized and readable.
  • Physics2D.OverlapCircle() - Professional ground detection. Creates an invisible circle at groundCheck position and checks if it touches anything on the groundLayer. More reliable than OnCollisionEnter.
  • LayerMask groundLayer - In Unity, create a "Ground" layer, assign it to platforms, then select it in the Inspector. This ensures only ground objects trigger isGrounded, not enemies or items.
  • Transform groundCheck - Create empty child GameObject, position at player's feet, drag into this field. The circle check happens at this position.

Top-Down 8-Direction Movement

Full movement in all directions with sprint functionality

Key Definitions
  • Top-Down View - Camera looks down at the game world from above (like classic Zelda)
  • 8-Direction Movement - Movement in 4 cardinal directions + 4 diagonal directions
  • Sprint/Run - Temporary speed boost while holding a modifier key (Shift)
  • Gravity Scale = 0 - Disables gravity for top-down games where Y is forward, not up
C# TopDownController.cs 
using UnityEngine;

public class TopDownController : MonoBehaviour
{
    // === MOVEMENT SETTINGS ===
    [Header("Movement")]
    public float moveSpeed = 5f;            // Normal walking speed
    public float sprintMultiplier = 1.5f;  // Sprint speed = moveSpeed √ó this
    
    // === PRIVATE VARIABLES ===
    private Rigidbody2D rb;                  // Physics component
    private Vector2 movement;                // Movement direction
    private float currentSpeed;              // Active speed (walk or sprint)

    void Start()
    {
        // Get Rigidbody2D component
        rb = GetComponent<Rigidbody2D>();
        
        // IMPORTANT: Disable gravity for top-down games
        // In top-down view, Y-axis is forward/backward, not up/down
        rb.gravityScale = 0f;
    }

    void Update()
    {
        // === GET INPUT ===
        // Supports WASD, Arrow keys, and gamepad
        movement.x = Input.GetAxisRaw("Horizontal");  // Left/Right
        movement.y = Input.GetAxisRaw("Vertical");    // Up/Down
        
        // === NORMALIZE ===
        // Ensures diagonal movement isn't faster
        movement = movement.normalized;
        
        // === SPRINT CHECK ===
        // Ternary operator: condition ? valueIfTrue : valueIfFalse
        // If Shift held: speed = moveSpeed √ó 1.5
        // If Shift not held: speed = moveSpeed
        currentSpeed = Input.GetKey(KeyCode.LeftShift) 
            ? moveSpeed * sprintMultiplier 
            : moveSpeed;
    }

    void FixedUpdate()
    {
        // === APPLY MOVEMENT ===
        rb.velocity = movement * currentSpeed;
    }
}
Code Explanation - Top-Down Specifics
  • rb.gravityScale = 0f; - Critical for top-down games! Without this, player would "fall" toward the bottom of the screen. Setting to 0 removes gravity entirely.
  • sprintMultiplier = 1.5f - Multiplier for sprint speed. Normal speed = 5, Sprint speed = 5 √ó 1.5 = 7.5 units/second.
  • ? : (Ternary Operator) - Shorthand if/else. condition ? A : B means "if condition is true, use A; otherwise use B". Cleaner than full if/else for simple choices.
  • Input.GetKey(KeyCode.LeftShift) - Returns true while Left Shift is HELD (not GetKeyDown). Player must hold Shift to maintain sprint.
