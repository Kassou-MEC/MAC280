What is C++?

C++ is a powerful, high-performance language developed by Bjarne Stroustrup at Bell Labs in 1979. It extends C with object-oriented features.

‚ö°

Why Learn C++?

C++ powers games, operating systems, browsers, and embedded systems. Understanding C++ gives you deep insight into how computers work.

Switch Statement

switch.cpp 
#include <iostream>
using namespace std;

int main() {
    int dayNum = 3;
    
    switch (dayNum) {
        case 1: cout << "Monday" << endl; break;
        case 2: cout << "Tuesday" << endl; break;
        case 3: cout << "Wednesday" << endl; break;
        case 4: cout << "Thursday" << endl; break;
        case 5: cout << "Friday" << endl; break;
        default: cout << "Weekend!" << endl;
    }
    return 0;
}
‚ö†Ô∏è Important

Don't forget the break statement! Without it, execution "falls through" to the next case.

Variable Scope

scope.cpp 
#include <iostream>
using namespace std;

int globalVar = 100;  // Global - accessible everywhere

void testScope() {
    int localVar = 50;  // Local to this function
    cout << "In function - global: " << globalVar << endl;
    cout << "In function - local: " << localVar << endl;
}

int main() {
    cout << "In main - global: " << globalVar << endl;
    testScope();
    
    { // Block scope
        int blockVar = 10;
        cout << "In block: " << blockVar << endl;
    }
    // blockVar not accessible here!
    
    return 0;
}

Pass by Value vs Reference

parameters.cpp 
#include <iostream>
using namespace std;

void addByValue(int num) { num += 10; }
void addByRef(int &num) { num += 10; }

int main() {
    int x = 5;
    
    addByValue(x);
    cout << "After pass by value: " << x << endl; // Still 5

    cout << "After pass by reference: " << x << endl; // Now 15
    
    return 0;
}

Introduction to Classes

classes.cpp 
#include <iostream>
using namespace std;

class BankAccount {
private:
    string owner;
    double balance;

public:
    BankAccount(string name, double initial) {
        owner = name;
        balance = initial;
    }
    
    void deposit(double amount) {
        balance += amount;
        cout << "Deposited $" << amount << endl;
    }
    
    bool withdraw(double amount) {
        if (amount <= balance) {
            balance -= amount;
            cout << "Withdrew $" << amount << endl;
            return true;
        }
        cout << "Insufficient funds!" << endl;
        return false;
    }
    
    double getBalance() { return balance; }
};

int main() {
    BankAccount acct("Alice", 1000);
    acct.deposit(500);
    acct.withdraw(200);
    cout << "Balance: $" << acct.getBalance() << endl;
    return 0;
}
üìò Struct vs Class

struct members are public by default, while class members are private by default. Classes emphasize encapsulation and data hiding.

Linear Search

linear_search.cpp 
#include <iostream>
using namespace std;

int linearSearch(int arr[], int size, int target) {
    for (int i = 0; i < size; i++) {
        if (arr[i] == target) return i;
    }
    return -1;
}

int main() {
    int nums[] = {64, 34, 25, 12, 22, 11, 90};
    int idx = linearSearch(nums, 7, 22);
    cout << "Found 22 at index: " << idx << endl;
    return 0;
}

Bubble Sort

bubble_sort.cpp 
#include <iostream>
using namespace std;

void bubbleSort(int arr[], int n) {
    for (int i = 0; i < n - 1; i++) {
        for (int j = 0; j < n - i - 1; j++) {
            if (arr[j] > arr[j + 1]) {
                int temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }
}

int main() {
    int arr[] = {64, 34, 25, 12, 22, 11, 90};
    int n = 7;
    
    cout << "Original: ";
    for(int i = 0; i < n; i++) cout << arr[i] << " ";
    
    bubbleSort(arr, n);
    
    cout << "\nSorted: ";
    for(int i = 0; i < n; i++) cout << arr[i] << " ";
    
    return 0;
}

Linear Search: O(n) ‚Äî checks each element once
